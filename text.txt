--// Gun Model local function createGunModel() local model = Instance.new("Model") model.Name = "GunModel"

local body = Instance.new("Part")
body.Name = "Body"
body.Size = Vector3.new(1.6, 0.5, 3.8)
body.Color = Color3.fromRGB(15, 15, 15)
body.Material = Enum.Material.Metal
body.Anchored = false
body.CanCollide = false
body.Parent = model

local grip = Instance.new("Part")
grip.Size = Vector3.new(0.6, 0.85, 1.1)
grip.Color = Color3.fromRGB(95, 52, 16)
grip.Material = Enum.Material.Wood
grip.Anchored = false
grip.CanCollide = false
grip.Parent = model

local barrel = Instance.new("Part")
barrel.Size = Vector3.new(0.35, 0.35, 2.2)
barrel.Color = Color3.fromRGB(10, 10, 10)
barrel.Material = Enum.Material.Metal
barrel.Anchored = false
barrel.CanCollide = false
barrel.Parent = model

local function weld(p0, p1)
	local w = Instance.new("WeldConstraint")
	w.Part0 = p0
	w.Part1 = p1
	w.Parent = p0
end
weld(body, grip)
weld(body, barrel)
grip.CFrame = body.CFrame * CFrame.new(0, -0.52, 0.8)
barrel.CFrame = body.CFrame * CFrame.new(0, 0, -2.5)
model.PrimaryPart = body
return model

end
